 34 "use strict";
 35 (function($) {
 36 
 37     var pluginName = 'jui_theme_switch',
 38         pluginStatus = 'jui_theme_switch_status';
 39 
 40     /* public methods ------------------------------------------------------- */
 41     var methods = {
 42 
 43         /**
 44          * @lends $.fn.jui_theme_switch
 45          */
 46         init: function(options) {
 47 
 48             var elem = this;
 49 
 50             return this.each(function() {
 51 
 52                 /**
 53                  * settings and defaults
 54                  * using $.extend, settings modification will affect elem.data() and vive versa
 55                  */
 56                 var settings = elem.data(pluginName);
 57                 if(typeof(settings) == 'undefined') {
 58                     var defaults = elem.jui_theme_switch('getDefaults');
 59                     settings = $.extend({}, defaults, options);
 60                 } else {
 61                     settings = $.extend({}, settings, options);
 62                 }
 63                 elem.data(pluginName, settings);
 64 
 65                 // initialize plugin status
 66                 if(typeof  elem.data(pluginStatus) === 'undefined') {
 67                     elem.data(pluginStatus, {});
 68                     elem.data(pluginStatus)["selected_theme"] = {};
 69                 }
 70 
 71 
 72                 elem.unbind("onChangeTheme").bind("onChangeTheme", settings.onChangeTheme);
 73                 elem.unbind("onDisplay").bind("onDisplay", settings.onDisplay);
 74 
 75                 var container_id = elem.attr("id"),
 76                     themes_len, i, html = '',
 77                     switcher_id = create_id(settings.switcher_id_prefix, container_id),
 78                     switcher_label_id = create_id(settings.switcher_label_id_prefix, container_id),
 79                     elem_switcher, elem_switcher_label,
 80                     elem_link = $("#" + settings.stylesheet_link_id),
 81                     selected = '', current_group, themes_group = '',
 82                     project_url = '';
 83 
 84                 $.ajax({
 85                     url: settings.datasource_url,
 86                     dataType: 'json',
 87                     success: function(data) {
 88 
 89                         themes_len = data.length;
 90 
 91                         html += '<label id="' + switcher_label_id + '" for="' + switcher_id + '">' + settings.switcher_label + '</label>';
 92                         html += '<select id="' + switcher_id + '" size="' + settings.list_size + '">';
 93                         for(i = 0; i < themes_len; i++) {
 94 
 95                             if(current_group = data[i]["active"] == "yes" || settings.show_all == "yes") {
 96 
 97                                 if(settings.use_groups == "yes") {
 98                                     current_group = data[i]["group"];
 99                                     if(current_group !== themes_group) {
100                                         html += '<optgroup label="' + current_group + '">';
101                                         themes_group = current_group;
102                                     }
103                                 }
104 
105                                 if(settings.default_theme == data[i]["theme_name"]) {
106                                     elem.data(pluginStatus)["selected_theme"] = data[i];
107                                     selected = ' selected="selected"';
108                                 } else {
109                                     selected = '';
110                                 }
111 
112                                 project_url = '';
113                                 if(data[i].hasOwnProperty("hosted_locally")) {
114                                     if(data[i]["hosted_locally"] == "yes") {
115                                         project_url = settings.project_url
116                                     }
117                                 }
118 
119                                 html += '<option value="' + project_url + data[i]["theme_url"] + '"' + selected + '>';
120                                 html += data[i]["theme_name"];
121                                 html += '</option>';
122 
123                                 if(settings.use_groups == "yes") {
124                                     if(i < themes_len - 1 && data[parseInt(i) + 1]["group"] !== themes_group) {
125                                         html += '</optgroup>';
126                                     }
127                                 }
128                             }
129                         }
130                         html += '</select>';
131 
132                         elem.html(html);
133 
134                         elem_switcher_label = $("#" + switcher_label_id);
135                         elem_switcher = $("#" + switcher_id);
136 
137                         elem_switcher_label.removeClass().addClass(settings.labelClass);
138                         elem_switcher.removeClass().addClass(settings.listClass);
139                         elem.removeClass().addClass(settings.containerClass);
140 
141                         // change theme
142                         elem.off('change', elem_switcher).on('change', elem_switcher, function() {
143                             elem_link.attr("href", elem_switcher.val());
144 
145                             for(i = 0; i < themes_len; i++) {
146                                 if($("#" + switcher_id + " option:selected").text() == data[i]["theme_name"]) {
147                                     elem.data(pluginStatus)["selected_theme"] = data[i];
148                                     break;
149                                 }
150                             }
151 
152                             // trigger event onChangeTheme
153                             elem.triggerHandler("onChangeTheme", data[i]);
154 
155                         });
156 
157 
158                         // JUI_THEME_SWITCH completed
159                         // trigger event onDisplay
160                         elem.triggerHandler("onDisplay");
161 
162                     }
163                 });
164 
165             });
166 
167         },
168 
169         /**
170          * Get default values
171          * @example $(element).jui_theme_switch('getDefaults');
172          * @return {Object}
173          */
174         getDefaults: function() {
175             return {
176                 switcher_label: "Select theme",
177                 default_theme: "ui-lightness",
178                 list_size: "1",
179                 use_groups: "yes",
180                 show_all: "no", // default is show items having "active": "yes"
181 
182                 project_url: "", // url relative to web server document root (required only for locally hosted themes)
183 
184                 containerClass: "switcher_container",
185                 labelClass: "switcher_label",
186                 listClass: "switcher_list",
187 
188                 switcher_label_id_prefix: "lbl_",
189                 switcher_id_prefix: "switcher_",
190 
191                 onChangeTheme: function() {
192 
193                 },
194                 onDisplay: function() {
195 
196                 }
197             };
198         },
199 
200         /**
201          * Get any option set to plugin using its name (as string)
202          * @example $(element).jui_theme_switch('getOption', some_option);
203          * @param {String} opt
204          * @return {*}
205          */
206         getOption: function(opt) {
207             var elem = this;
208             return elem.data(pluginName)[opt];
209         },
210 
211         /**
212          * Get all options
213          * @example $(element).jui_theme_switch('getAllOptions');
214          * @return {*}
215          */
216         getAllOptions: function() {
217             var elem = this;
218             return elem.data(pluginName);
219         },
220 
221         /**
222          * Set option
223          * @example $(element).jui_theme_switch('setOption', 'option_name',  'option_value',  reinit);
224          * @param {String} opt option names
225          * @param val
226          * @param {Boolean} reinit
227          */
228         setOption: function(opt, val, reinit) {
229             var elem = this;
230             elem.data(pluginName)[opt] = val;
231             if(reinit) {
232                 elem.jui_theme_switch('init');
233             }
234         },
235 
236         /**
237          * Refresh plugin
238          * @example $(element).jui_theme_switch('refresh');
239          * @return {*|jQuery}
240          */
241         refresh: function() {
242             var elem = this;
243             elem.jui_theme_switch();
244         },
245 
246         /**
247          * Destroy plugin
248          * @example $(element).jui_theme_switch('destroy');
249          * @return {*|jQuery}
250          */
251         destroy: function() {
252             return $(this).each(function() {
253                 var $this = $(this);
254 
255                 $this.removeData(pluginName);
256             });
257         },
258 
259         /**
260          * Get selected theme (as JSON object)
261          *
262          * @example $(element).jui_theme_switch('getTheme');
263          * @return {*}
264          */
265         getTheme: function() {
266             var elem = this;
267             return elem.data(pluginStatus)["selected_theme"];
268         }
269     };

278     /**
279      * Create element id
280      * @param prefix
281      * @param plugin_container_id
282      * @return {*}

284     var create_id = function(prefix, plugin_container_id) {
285         return prefix + plugin_container_id;
286     };

288     /**
289      * jui_theme_switch - short description.
290      *
291      * @class jui_theme_switch
292      * @memberOf $.fn
293      */
294     $.fn.jui_theme_switch = function(method) {
295 
296         // OPTIONAL
297         if(this.size() != 1) {
298             var err_msg = 'You must use this plugin (' + pluginName + ') with a unique element (at once)';
299             this.html('<span style="color: red;">' + 'ERROR: ' + err_msg + '</span>');
300             $.error(err_msg);
301         }
302 
303         // Method calling logic
304         if(methods[method]) {
305             return methods[ method ].apply(this, Array.prototype.slice.call(arguments, 1));
306         } else if(typeof method === 'object' || !method) {
307             return methods.init.apply(this, arguments);
308         } else {
309             $.error('Method ' + method + ' does not exist on jQuery.' + pluginName);
310         }
311 
312     };

314 })(jQuery);
