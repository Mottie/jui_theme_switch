 34 "use strict";
 35 (function($) {
 36 
 37     var pluginName = 'jui_theme_switch';
 38 
 39     /* public methods ------------------------------------------------------- */
 40     var methods = {
 41 
 42         /**
 43          * @lends $.fn.jui_theme_switch
 44          */
 45         init: function(options) {
 46 
 47             var elem = this;
 48 
 49             return this.each(function() {
 50 
 51                 /**
 52                  * settings and defaults
 53                  * using $.extend, settings modification will affect elem.data() and vive versa
 54                  */
 55                 var settings = elem.data(pluginName);
 56                 if(typeof(settings) == 'undefined') {
 57                     var defaults = elem.jui_theme_switch('getDefaults');
 58                     settings = $.extend({}, defaults, options);
 59                 } else {
 60                     settings = $.extend({}, settings, options);
 61                 }
 62                 elem.data(pluginName, settings);
 63 
 64                 var container_id = elem.attr("id"),
 65                     themes_len, i, html = '',
 66                     switcher_id = create_id(settings.switcher_id_prefix, container_id),
 67                     switcher_label_id = create_id(settings.switcher_label_id_prefix, container_id),
 68                     elem_switcher, elem_switcher_label,
 69                     elem_link = $("#" + settings.stylesheet_link_id),
 70                     selected = '', current_group, themes_group = '';
 71 
 72                 $.ajax({
 73                     url: settings.datasource_url,
 74                     dataType: 'json',
 75                     success: function(data) {
 76 
 77                         themes_len = data.length;
 78 
 79                         html += '<label id="' + switcher_label_id + '" for="' + switcher_id + '">' + settings.switcher_label + '</label>';
 80                         html += '<select id="' + switcher_id + '" size="' + settings.list_size + '">';
 81                         for(i = 0; i < themes_len; i++) {
 82 
 83                             if(current_group = data[i]["active"] == "yes" || settings.show_all == "yes") {
 84 
 85                                 if(settings.use_groups == "yes") {
 86                                     current_group = data[i]["group"];
 87                                     if(current_group !== themes_group) {
 88                                         html += '<optgroup label="' + current_group + '">';
 89                                         themes_group = current_group;
 90                                     }
 91                                 }
 92 
 93                                 selected = (settings.default_theme == data[i]["theme_name"] ? ' selected="selected"' : '');
 94                                 html += '<option value="' + data[i]["theme_url"] + '"' + selected + '>';
 95                                 html += data[i]["theme_name"];
 96                                 html += '</option>';
 97 
 98                                 if(settings.use_groups = "yes") {
 99                                     if(i < themes_len - 1 && data[parseInt(i) + 1]["group"] !== themes_group) {
100                                         html += '</optgroup>';
101                                     }
102                                 }
103                             }
104                         }
105                         html += '</select>';
106 
107                         elem.html(html);
108 
109                         elem_switcher_label = $("#" + switcher_label_id);
110                         elem_switcher = $("#" + switcher_id);
111 
112                         elem_switcher_label.removeClass().addClass(settings.labelClass);
113                         elem_switcher.removeClass().addClass(settings.listClass);
114                         elem.removeClass().addClass(settings.containerClass);
115 
116                         // change theme
117                         elem.off('change', elem_switcher).on('change', elem_switcher, function() {
118                             elem_link.attr("href", elem_switcher.val());
119                         });
120 
121                     }
122                 });
123 
124             });
125 
126         },
127 
128         /**
129          * Get default values
130          * @example $(element).jui_theme_switch('getDefaults');
131          * @return {Object}
132          */
133         getDefaults: function() {
134             return {
135                 switcher_label: "Select theme",
136                 default_theme: "ui-lightness",
137                 list_size: "1",
138                 use_groups: "yes",
139                 show_all: "no", // default is show items having "active": "yes"
140 
141                 switcher_label_id_prefix: "lbl_",
142                 switcher_id_prefix: "switcher_"
143             };
144         },
145 
146         /**
147          * Get any option set to plugin using its name (as string)
148          * @example $(element).jui_theme_switch('getOption', some_option);
149          * @param {String} opt
150          * @return {*}
151          */
152         getOption: function(opt) {
153             var elem = this;
154             return elem.data(pluginName)[opt];
155         },
156 
157         /**
158          * Get all options
159          * @example $(element).jui_theme_switch('getAllOptions');
160          * @return {*}
161          */
162         getAllOptions: function() {
163             var elem = this;
164             return elem.data(pluginName);
165         },
166 
167         /**
168          * Set option
169          * @example $(element).jui_theme_switch('setOption', 'option_name',  'option_value',  reinit);
170          * @param {String} opt option names
171          * @param val
172          * @param {Boolean} reinit
173          */
174         setOption: function(opt, val, reinit) {
175             var elem = this;
176             elem.data(pluginName)[opt] = val;
177             if(reinit) {
178                 elem.jui_theme_switch('init');
179             }
180         },
181 
182         /**
183          * Refresh plugin
184          * @example $(element).jui_theme_switch('refresh');
185          * @return {*|jQuery}
186          */
187         refresh: function() {
188             var elem = this;
189             elem.jui_theme_switch();
190         },
191 
192         /**
193          * Destroy plugin
194          * @example $(element).jui_theme_switch('destroy');
195          * @return {*|jQuery}
196          */
197         destroy: function() {
198             return $(this).each(function() {
199                 var $this = $(this);
200 
201                 $this.removeData(pluginName);
202             });
203         }
204     };
205 
206     /* private methods ------------------------------------------------------ */
207     /** the foolowing is OPTIONAL in case private methods will be documented  */

213     /**
214      * Create element id
215      * @param prefix
216      * @param plugin_container_id
217      * @return {*}
218      */
219     var create_id = function(prefix, plugin_container_id) {
220         return prefix + plugin_container_id;
221     };
222 
223     /**
224      * jui_theme_switch - short description.
225      *
226      * @class jui_theme_switch
227      * @memberOf $.fn
228      */
229     $.fn.jui_theme_switch = function(method) {
230 
231         // OPTIONAL
232         if(this.size() != 1) {
233             var err_msg = 'You must use this plugin (' + pluginName + ') with a unique element (at once)';
234             this.html('<span style="color: red;">' + 'ERROR: ' + err_msg + '</span>');
235             $.error(err_msg);
236         }
237 
238         // Method calling logic
239         if(methods[method]) {
240             return methods[ method ].apply(this, Array.prototype.slice.call(arguments, 1));
241         } else if(typeof method === 'object' || !method) {
242             return methods.init.apply(this, arguments);
243         } else {
244             $.error('Method ' + method + ' does not exist on jQuery.' + pluginName);
245         }
246 
247     };
248 
249 })(jQuery);
